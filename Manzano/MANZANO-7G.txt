   Escreva "Soma B + D: ", B + D
   Escreva "Soma C + D: ", C + D

   Escreva "Multiplicação A * D: ", A * D
   Escreva "Multiplicação B * C: ", B * C
   Escreva "Multiplicação B * D: ", B * D
