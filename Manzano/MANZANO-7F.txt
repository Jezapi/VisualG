// Data atual   : 22/10/2024

Algoritmo "TrocarValores"
   // Declaração de variáveis
   Declare A, B, Temp: Real
   // Entrada de dados
   Escreva "Digite o valor de A:"
   Leia A
   Escreva "Digite o valor de B:"
   Leia B
   // Troca dos valores
   Temp ? A
   A ? B
   B ? Temp
   // Saída de dados
   Escreva "Valor de A após a troca: ", A
   Escreva "Valor de B após a troca: ", B
FimAlgoritmo
