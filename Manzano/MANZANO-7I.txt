Algoritmo "QuadradoDaDiferenca"
   // Declaração de variáveis
   Declare A, B, Resultado: Inteiro
   // Entrada de dados
   Escreva "Digite o primeiro valor (A):"
   Leia A
   Escreva "Digite o segundo valor (B):"
   Leia B
   // Cálculo do quadrado da diferença
   Resultado ? (A - B) * (A - B)
   // Saída de dados
   Escreva "O quadrado da diferença de A e B é: ", Resultado
FimAlgoritmo
