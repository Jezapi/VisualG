Algoritmo "CustoDasMacas"
// Disciplina   : Linguagem e Lógica de Programação
// Professor    : Obama
// Descrição    : Calcular o custo total da compra de maçãs, considerando o preço com desconto para 12 ou mais unidades.
// Autor(a)     : Jenifer Souza Pimentel Pais
// Data atual   : 22/10/2024

Var
   macas, custo: inteiro

Inicio
   Escreva("Digite o número de maçãs compradas: ")
   Leia(macas)

   Se macas >= 12 Entao
      custo := macas * 1
   Senao
      custo := macas * 1.30
   FimSe

   Escreva("O custo total da compra é: R$ ", custo)
Fimalgoritmo
