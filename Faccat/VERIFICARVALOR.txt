Var
   valor: real

Inicio
    // Entrada de dados
    Escreva("Informe um valor: ")
    Leia(valor)

    // Verificação do valor
    Se valor > 0 entao
        Escreva("Positivo")
    SenaoSe valor < 0 entao
        Escreva("Negativo")
    Senao
        Escreva("Zero")
    FimSe
Fimalgoritmo
